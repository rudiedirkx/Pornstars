#
# Tabel structuur voor tabel `alliances`
#
CREATE TABLE alliances (
  id tinyint(4) NOT NULL auto_increment,
  tag varchar(8) NOT NULL default '',
  pwd varchar(255) NOT NULL default '',
  name varchar(40) NOT NULL default '',
  leader_id tinyint(4) NOT NULL default '0',
  score bigint(20) NOT NULL default '0',
  size bigint(20) NOT NULL default '0',
  num_members int(11) NOT NULL default '0',
  PRIMARY KEY  (id)
) TYPE=MyISAM;


#
# Tabel structuur voor tabel `fleets`
#
CREATE TABLE fleets (
  id bigint(20) NOT NULL auto_increment,
  owner_id smallint(6) NOT NULL default '0',
  owner_x smallint(6) NOT NULL default '0',
  owner_y smallint(6) NOT NULL default '0',
  destination_id smallint(6) NOT NULL default '0',
  destination_x smallint(6) NOT NULL default '0',
  destination_y smallint(6) NOT NULL default '0',
  eta tinyint(4) NOT NULL default '0',
  starteta tinyint(4) NOT NULL default '0',
  purpose enum('','attack','defend','return') NOT NULL default '',
  actiontime tinyint(4) NOT NULL default '0',
  startactiontime tinyint(4) NOT NULL default '0',
  fleetname enum('base','1','2') NOT NULL default 'base',
  infinitys bigint(20) NOT NULL default '0',
  wraiths bigint(20) NOT NULL default '0',
  warfrigs bigint(20) NOT NULL default '0',
  astropods bigint(20) NOT NULL default '0',
  cobras bigint(20) NOT NULL default '0',
  destroyers bigint(20) NOT NULL default '0',
  scorpions bigint(20) NOT NULL default '0',
  antennas bigint(20) NOT NULL default '0',
  PRIMARY KEY  (id)
) TYPE=MyISAM COMMENT='Elke player kan max 3 fleets hebben, waarvan 1 _base';


#
# Tabel structuur voor tabel `galaxies`
#
CREATE TABLE galaxies (
  id tinyint(4) NOT NULL auto_increment,
  x tinyint(4) NOT NULL default '0',
  score bigint(20) NOT NULL default '0',
  size bigint(20) NOT NULL default '0',
  galname varchar(255) NOT NULL default '',
  galpic varchar(255) NOT NULL default '',
  gcmessage text NOT NULL,
  num_planets tinyint(4) NOT NULL default '0',
  PRIMARY KEY  (id)
) TYPE=MyISAM;


#
# Tabel structuur voor tabel `logbook`
#
CREATE TABLE logbook (
  id bigint(20) NOT NULL auto_increment,
  uid int(11) NOT NULL default '0',
  action varchar(20) NOT NULL default '',
  time bigint(20) NOT NULL default '0',
  myt bigint(20) NOT NULL default '0',
  text text NOT NULL,
  ip varchar(20) NOT NULL default '',
  PRIMARY KEY  (id)
) TYPE=MyISAM;


#
# Tabel structuur voor tabel `mail`
#
CREATE TABLE mail (
  id int(11) NOT NULL auto_increment,
  uid int(11) NOT NULL default '0',
  afzender mediumint(9) NOT NULL default '0',
  time bigint(20) NOT NULL default '0',
  myt int(11) NOT NULL default '0',

  seen enum('0','1') NOT NULL default '0',
  visibility enum('1','0') NOT NULL default '1',

) TYPE=MyISAM;


#

#
# Tabel structuur voor tabel `prefs`
#
CREATE TABLE prefs (
  id tinyint(1) NOT NULL auto_increment,
  gamename varchar(24) NOT NULL default '',
  tickcount int(11) NOT NULL default '0',
  military_attack enum('1','0') NOT NULL default '1',
  military_defend enum('1','0') NOT NULL default '1',
  military_scorelimit tinyint(2) NOT NULL default '0',
  tickertime mediumint(9) NOT NULL default '20',
  general_login enum('1','0') NOT NULL default '1',
  general_signup enum('1','0') NOT NULL default '1',
  general_adminmsg text NOT NULL,
  pwd_voor_ticker enum('1','0') NOT NULL default '1',
  ticker_on enum('1','0') NOT NULL default '1',
  general_gamestoptick bigint(20) NOT NULL default '0',
  autologout enum('0','1') NOT NULL default '0',
  PRIMARY KEY  (id)
) TYPE=MyISAM;


#
# Tabel structuur voor tabel `production`
#
CREATE TABLE production (
  id bigint(20) NOT NULL auto_increment,
  uid int(11) NOT NULL default '0',
  unit enum('infinitys','wraiths','warfrigs','astropods','cobras','destroyers','scorpions','antennas','rcannons','avengers','lstalkers','roidscans','sectorscans','unitscans','pduscans','militaryscans','newsscans','waveamps','waveblockers') NOT NULL default 'infinitys',
  eta tinyint(4) NOT NULL default '0',
  number bigint(20) NOT NULL default '0',
  u_of_w enum('u','w') NOT NULL default 'u',
  PRIMARY KEY  (id)
) TYPE=MyISAM;
